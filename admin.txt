-- Scripts:

local function YDGD_fake_script() -- mainframe.LocalScript 
	local script = Instance.new('LocalScript', mainframe)

	-- Put this LocalScript inside the Frame
	local frame = script.Parent
	
	-- set the starting position
	frame.Position = UDim2.new(1, 0, 0.828, 0)
	
	-- how far to move
	local moveLeft = UDim2.new(0, -300, 0, 0) 
	local moveRight = UDim2.new(0, 500, 0, 0) 
	
	-- how fast (seconds per tween)
	local tweenTime = 1 
	
	local TweenService = game:GetService("TweenService")
	
	-- function to tween the frame to a target offset
	local function moveFrame(offset)
		local goal = {}
		goal.Position = frame.Position + offset
	
		local tweenInfo = TweenInfo.new(
			tweenTime,
			Enum.EasingStyle.Quad,
			Enum.EasingDirection.Out
		)
	
		local tween = TweenService:Create(frame, tweenInfo, goal)
		tween:Play()
		tween.Completed:Wait()
	end
	
	-- run the sequence
	task.wait(1) -- small delay after loading
	moveFrame(moveLeft)   -- move left 300px
	task.wait(3)          -- wait 3 seconds
	moveFrame(moveRight)  -- move right 500px
	
end
coroutine.wrap(YDGD_fake_script)()
